Функция Получить(пЗначение) Экспорт
	
	Если ТипЗнч(пЗначение) = Тип("СправочникСсылка.Контрагенты") Тогда
		Текст = "ВЫБРАТЬ Организация КАК Значение ИЗ РегистрСведений.СобственныеКонтрагенты ГДЕ Контрагент = &Значение";
		ПоУмолчанию = Справочники.Организации.ПустаяСсылка();
	ИначеЕсли ТипЗнч(пЗначение) = Тип("СправочникСсылка.Организации") Тогда
		Текст = "ВЫБРАТЬ Контрагент КАК Значение ИЗ РегистрСведений.СобственныеКонтрагенты ГДЕ Организация = &Значение";
		ПоУмолчанию = Справочники.Контрагенты.ПустаяСсылка();
	КонецЕсли;
	
	Запрос = Новый Запрос(Текст);
	Запрос.УстановитьПараметр("Значение" , пЗначение);
	Результат = Запрос.Выполнить();
	
	Если Результат.Пустой() Тогда
		Возврат ПоУмолчанию;
	Иначе
		Выборка = Результат.Выбрать();
		Выборка.Следующий();
		Возврат Выборка.Значение;
	КонецЕсли;
	
КонецФункции
