
&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	Если Параметры.Свойство("Проведен") Тогда
		ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Проведен", Параметры.Проведен);
	КонецЕсли;
	
	Если Параметры.Свойство("Отправитель") Тогда
		ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Отправитель", Параметры.Отправитель);
	КонецЕсли;
	
	Если Параметры.Свойство("Получатель") Тогда
		ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Получатель", Параметры.Получатель);
	КонецЕсли;
	
	Если Параметры.Свойство("Статус") Тогда
		Если ТипЗнч(Параметры.Статус) = Тип("СписокЗначений") Тогда
			ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Статус", Параметры.Статус, ВидСравненияКомпоновкиДанных.ВСписке);
		Иначе
			ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Статус", Параметры.Статус);
		КонецЕсли;
	КонецЕсли;
	
	Если Параметры.Свойство("Тип") Тогда
		Если ТипЗнч(Параметры.Статус) = Тип("СписокЗначений") Тогда
			ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Тип", Параметры.Тип, ВидСравненияКомпоновкиДанных.ВСписке);
		Иначе
			ВетисОбщегоНазначения.ДобавитьОтбор(Список.Отбор, "Тип", Параметры.Тип);
		КонецЕсли;
	КонецЕсли;
	
КонецПроцедуры
