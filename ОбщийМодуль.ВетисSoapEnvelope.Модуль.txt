
Функция Envelope(пФабрика, пОбъект = Неопределено, пИмя = Неопределено) Экспорт
	
	_Объект = пФабрика.Создать(URI(), "Envelope");
	
	_Объект.Body = Body(пФабрика, пОбъект, пИмя);
	
	Возврат _Объект;
	
КонецФункции

Функция Body(пФабрика, пОбъект = Неопределено, пИмя = Неопределено) Экспорт
	
	_Объект = пФабрика.Создать(URI(), "Body");
	
	Если НЕ пОбъект = Неопределено Тогда
		
		_Объект.Добавить(ФормаXML.Элемент, пОбъект.Тип().URIПространстваИмен, пИмя, пОбъект);
		
	КонецЕсли;
	
	Возврат _Объект;
	
КонецФункции

Функция Извлечь(пEnvelope) Экспорт
	
	Возврат пEnvelope.Body;
	
КонецФункции


Функция URI() Экспорт
	
	Возврат "http://schemas.xmlsoap.org/soap/envelope/";
