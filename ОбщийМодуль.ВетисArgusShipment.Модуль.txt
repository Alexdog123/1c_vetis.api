
// Тип, описывающий товарно-транспортную накладную.
Функция Waybill(type, issueNumber, issueDate, broker, transportInfo, transportStorageType) Экспорт
	
	_Объект = ФабрикаXDTO.Создать(ФабрикаXDTO.Тип(URI(), "Waybill"));
	
	_Объект.type                    = type;//Тип товарно-транспортной накладной.
	//_Объект.issueSeries             = ;//Серия документа.
	_Объект.issueNumber             = issueNumber;//Номер документа.
	_Объект.issueDate               = issueDate;//Дата документа.
	//_Объект.consignor               = ;//Сведения об отправителе (владельце).
	//_Объект.consignee               = ;//Сведения о получателе груза.
	_Объект.broker                  = broker;//Сведения о фирме-посреднике.
	_Объект.transportInfo           = transportInfo;//Сведения о транспорте.
	_Объект.transportStorageType    = transportStorageType;//Способ хранения при перевозке.
	//_Объект.cargoReloadingPointList = ;//Сведения о маршруте следования (пунктах перегрузки).
	
	Возврат _Объект;
	
КонецФункции

	_Объект.transportType   = transportType;  //Тип транспортного средства.

#Область TransportationStorageType
// Тип, описывающий cпособ хранения при перевозке.

Функция TransportationStorageType(type) Экспорт
	
	Если      type = 1 Тогда
		Возврат ВетисКонстанты.TransportationStorageType_FROZEN();
	ИначеЕсли type = 2 Тогда
		Возврат ВетисКонстанты.TransportationStorageType_CHILLED();
	ИначеЕсли type = 3 Тогда
		Возврат ВетисКонстанты.TransportationStorageType_COOLED();
	ИначеЕсли type = 4 Тогда
		Возврат ВетисКонстанты.TransportationStorageType_VENTILATED();
	Иначе
		Возврат Неопределено;
	КонецЕсли;
	
КонецФункции

Функция TransportationStorageTypeОписание(type) Экспорт
	
	Если      type = ВетисКонстанты.TransportationStorageType_FROZEN()  ИЛИ type = 1 Тогда
		Возврат "Замороженные";
	ИначеЕсли type = ВетисКонстанты.TransportationStorageType_CHILLED() ИЛИ type = 2 Тогда
		Возврат "Охлажденные";
	ИначеЕсли type = ВетисКонстанты.TransportationStorageType_COOLED()  ИЛИ type = 3 Тогда
		Возврат "Охлаждаемые";
	ИначеЕсли type = ВетисКонстанты.TransportationStorageType_VENTILATED() ИЛИ type = 4 Тогда
		Возврат "Вентилируемые";
	Иначе
		Возврат Неопределено;
	КонецЕсли;
	
КонецФункции

Функция WaybillTypeОписание(type) Экспорт

	ИначеЕсли type = 3 Тогда

	ИначеЕсли type = 4 Тогда
		Возврат "Авианакладная";
	Иначе
		Возврат Неопределено;
	КонецЕсли;
