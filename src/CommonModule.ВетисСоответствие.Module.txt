Функция ПолучитьСсылку(guid, пТип = Неопределено, ПоУмолчанию = Неопределено) Экспорт
	
	Запрос = Новый Запрос;
	Запрос.Текст =
	"ВЫБРАТЬ ПЕРВЫЕ 1
	|	_Таблица.Ссылка
	|ИЗ
	|	РегистрСведений.ВетисСоответствие КАК _Таблица
	|ГДЕ
	|	_Таблица.guid = &guid
	//|//1	И _Таблица.Ссылка ССЫЛКА {Тип}
	|//1	{Тип}
	|	И ИСТИНА";
	
	Запрос.УстановитьПараметр("guid", guid);
	
	Если НЕ пТип = Неопределено Тогда
		Запрос.Текст = СтрЗаменить(Запрос.Текст, "//1", "");
		_Типы = СтрРазделить(пТип, ",");
		_Отбор = "";
		Для каждого _Тип Из _Типы Цикл
			_Отбор = _Отбор + " ИЛИ _Таблица.Ссылка ССЫЛКА " + _Тип;
		КонецЦикла;
		Запрос.Текст = СтрЗаменить(Запрос.Текст, "{Тип}", "И (Ложь" + _Отбор + ")");
	КонецЕсли;
	
	Результат = Запрос.Выполнить();
	Выборка = Результат.Выбрать();
	
	Если Выборка.Следующий() Тогда
		Возврат Выборка.Ссылка;
	Иначе
		Возврат ПоУмолчанию;
	КонецЕсли;
	
КонецФункции
