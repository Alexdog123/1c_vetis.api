&НаСервере
Процедура СписокВетисЗаполнитьСервер()
	
	Перем _Item, _Параметры;
	
	_ВетисDictionaryService = ?(ВетисXSD.Версия_2_0(), ВетисDictionaryService_2_0, ВетисDictionaryService);
	
	СписокВетис.Очистить();
	
	Пока _ВетисDictionaryService.GetUnitListСледующий(_Item, _Параметры) Цикл
		
		// только базовые
		Если НЕ _Item.guid = _Item.commonUnitGuid Тогда
			Продолжить;
		КонецЕсли;
		
		ТаблицаСтрока = СписокВетис.Добавить();
		ТаблицаСтрока.name = _Item.name;
		ТаблицаСтрока.guid = _Item.guid;
		
	КонецЦикла;
	
	СписокВетис.Сортировать("name");

&НаСервере
Процедура СоздатьСвязьСервер(пСсылка) Экспорт
	
	_ТекДанные = СписокВетис.НайтиПоИдентификатору(Элементы.СписокВетис.ТекущаяСтрока);
	
	Если Элементы.СправочникСписок.ВыделенныеСтроки.Количество() > 0 Тогда
		Для Каждого _ТекДанные1с из Элементы.СправочникСписок.ВыделенныеСтроки Цикл
			
			ВетисСоответствие.Добавить(_ТекДанные1с.Ссылка, _ТекДанные.guid, _ТекДанные.name);
			
		КонецЦикла;
		
	Иначе
		ВетисСоответствие.Добавить(пСсылка, _ТекДанные.guid, _ТекДанные.name);
		
	КонецЕсли;
	
КонецПроцедуры
