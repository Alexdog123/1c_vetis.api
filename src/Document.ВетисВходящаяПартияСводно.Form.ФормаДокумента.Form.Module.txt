&НаКлиенте
Процедура КомандаОтправить(Команда)
	
	Если НЕ Объект.Проведен Тогда
		Возврат;
	КонецЕсли;
	
	_Индекс = 0;
	
	Для _Индекс = 0 По Объект.СкладскойЖурнал.Количество()-1 Цикл
		
		Если ЗначениеЗаполнено(Объект.СкладскойЖурнал[_Индекс].guid) Тогда
			Продолжить;
		КонецЕсли;
		
		КомандаОтправитьСервер(_Индекс);
		
	КонецЦикла;
	
КонецПроцедуры

&НаСервере
Процедура КомандаОтправитьСервер(пНомерСтроки)
	
	_Отказ = Ложь;
	
	_СтрокаСЖ = Объект.СкладскойЖурнал.Получить(пНомерСтроки);
	
	ВетисMercuryApplicationsСлой1с.IncomingOperation(_СтрокаСЖ, _Отказ);
	
	Если НЕ _Отказ = Ложь Тогда
		Модифицированность = Истина;
	КонецЕсли;
	
КонецПроцедуры
