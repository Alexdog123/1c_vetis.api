	_Отбор = Новый Структура("fullName",ВетисСтрокаПоиска);
	
	Пока Истина Цикл
		
		Попытка
			_List = ВетисCerberusEnterprise.GetBusinessEntityList(_Отбор, _ListOptions);
		Исключение
			ВетисЖурнал.ДобавитьИнформациюОбОшибке(ИнформацияОбОшибке());
			Прервать;
		КонецПопытки;

				Если ЗначениеЗаполнено(_Entity.fullName) Тогда
					_СписокВетисСтрока.name = _Entity.fullName;
				КонецЕсли;
			Иначе

			КонецЕсли;
			
			_СписокВетисСтрока.type = ВетисCerberusEnterprise.BusinessEntityTypeОписание(_Entity.type);
			
			_СписокВетисСтрока.address = _Entity.juridicalAddress.addressView;
			
		КонецЦикла;
