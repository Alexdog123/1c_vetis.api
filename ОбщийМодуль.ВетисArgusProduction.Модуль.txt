Функция Product(name, code, productType = 1, guid = Неопределено) Экспорт
	
	_Фабрика = Фабрика();
	
	_Объект = _Фабрика.Создать(URI(), "Product");
	
	_Объект.guid = guid;
	_Объект.name = name;// Название продукции.
	_Объект.code = code;// Код ТНВЭД.
	_Объект.englishName = "";
	_Объект.productType = productType;
	
	Возврат _Объект;
